\documentclass{standalone}
\usepackage{fontspec}
\usepackage[rgb,dvipsnames,svgnames]{xcolor}
\usepackage{calc}
%
\setmainfont[BoldFont={* Medium}]{Noto Serif}
\setsansfont[BoldFont={* Medium}]{Noto Sans}
\setmonofont{Fira Mono}
\defaultfontfeatures{MatchLowerCase}
%
\colorlet{border}{RosyBrown!50!Black}%
\colorlet{background}{CornflowerBlue}%
\colorlet{cogito}{Black!80!}%
\colorlet{ergo}{Black!50!}%
\colorlet{sum}{Black!20!}%
%
\begin{document}
%
% These units of length are absolute and 
% these definitions can appear anywhere.
% 
\setlength{\fboxrule}{0.5pt}% border thickness
\setlength{\fboxsep}{2.0pt}% all round distance between box and text
%
% These units of length are relative to font size
% and must appear in the same scope as the text, and 
% only after the font size has been selected.
% Here, the default font size for the class is selected 
% automatically.
% 
\newlength{\overlap}\setlength{\overlap}{-0.4em}% negative kern value
\newlength{\moveup}\setlength{\moveup}{0.2em}% vertical box raising/lowering
%
\newsavebox{\CESBox}%
\sbox{\CESBox}{% 
{\textcolor{cogito}{Cogito}}% First word
\hspace*{\overlap}\raisebox{\moveup}{\textcolor{ergo}{Ergo}}% Second word
\hspace*{\overlap}\raisebox{2\moveup}{\textcolor{sum}{Sum}}% Third word
}%
%
\fcolorbox{border}{background}{\usebox{\CESBox}}%
%
% The problem with this output is that the letter "S" of "Sum" is written 
% over the "o" of "Ergo".
% We try to overcome this in our second attempt at typesetting this logo.
% 
\end{document}
